  ```kt
      // 원래 forEach
      public inline fun <T> Iterable<T>.forEach(action: (T) -> Unit): Unit {
         for (element in this) action(element)
      }

      // 실제 구현한 재귀 활용 forEach
      fun LinkedList.forEach(node: ListNode?, action: (ListNode) -> Unit) {
        if (node == null) return
        action(node)
        forEach(node.next, action)
      }
  ```
